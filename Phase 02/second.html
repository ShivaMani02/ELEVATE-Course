2. Conditionals in JavaScript

2.1 if Statement

let age = 18;
if (age >= 18) {
  console.log("You are eligible to vote.");
}

2.2 if-else Statement

let num = 7;
if (num % 2 === 0) {
  console.log("Even number");
} else {
  console.log("Odd number");
}

2.3 if-else if-else Statement

let marks = 75;
if (marks >= 90) {
  console.log("Grade: A");
} else if (marks >= 75) {
  console.log("Grade: B");
} else if (marks >= 60) {
  console.log("Grade: C");
} else {
  console.log("Grade: F");
}

2.4 Ternary Operator

let age = 20;
let result = (age >= 18) ? "Adult" : "Minor";
console.log(result); // Adult

2.5 switch Statement

let day = 3;
switch (day) {
  case 1:
    console.log("Monday");
    break;
  case 2:
    console.log("Tuesday");
    break;
  case 3:
    console.log("Wednesday");
    break;
  default:
    console.log("Another day");
}

3. Loops in JavaScript

3.1 for Loop

for (let i = 1; i <= 5; i++) {
  console.log("Count:", i);
}

3.2 while Loop

let i = 1;
while (i <= 5) {
  console.log("While loop:", i);
  i++;
}

3.3 do...while Loop

let i = 1;
do {
  console.log("Do While loop:", i);
  i++;
} while (i <= 5);

3.4 for...of Loop (arrays)

let fruits = ["apple", "banana", "cherry"];
for (let fruit of fruits) {
  console.log(fruit);
}

3.5 for...in Loop (objects)

let person = {
  name: "Alice",
  age: 25,
  city: "New York"
};
for (let key in person) {
  console.log(key + ": " + person[key]);
}
